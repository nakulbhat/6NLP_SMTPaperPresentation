\begin{frame}{\faRandom\ System Combination Task}
\begin{itemize}
    \item Participants were allowed to use \textbf{Moses}, an open-source
        translation toolkit.
    \item Moses performed at par with the best submissions of the previous year.
    \item 22 groups from 20 institutions submitted; 12 groups used the Moses toolkit.
    \item 87 systems were submitted, and an additional 42 were submitted as secondary systems. 6 combination systems were also submitted.
\end{itemize}

    \begin{block}{System Combination}
    \begin{itemize}
        \item Na\"ive Combination
        \item Multi-source combination
    \end{itemize}
\end{block}
\end{frame}

\begin{frame}{\faTimesCircle\ Evaluation Task}
    \begin{itemize}
        \item Outputs from select systems were evaluated by human annotators.
            These were split into two types:
            \begin{enumerate}
                \item Ranked lists
                \item Edited translations
            \end{enumerate}
        \item Automated metrics also scored the translations.
        \item Both scores were compared to find the best automated metrics.
    \end{itemize}
\end{frame}
