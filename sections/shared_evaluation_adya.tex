\begin{frame}{System-Level Correlation}
\begin{itemize}
\item Compare metric rankings with human rankings
\item Use Spearman rank correlation ($\rho$)
    \vspace{-0.5em}
        \begin{equation*}
            \rho = 1 - \frac{6 \sum d_{i}^2}{n (n^2 - 1)}
        \end{equation*}
    \vspace{-0.5em}
\begin{tabular}{lll}
    where, &&\\
    & $d_i$ & Difference between the rank for system $i$ \\
    & $n$ & Number of systems \\
\end{tabular}
\item Convert system scores into ranks
\item Measure similarity with human judgments
\end{itemize}
\end{frame}

\begin{frame}{Sentence-Level Consistency}
\begin{itemize}
\item Compare metrics at sentence level
\item Use pairwise system comparisons
\item Check agreement with human judgments
\item Report percentage of consistent decisions
\end{itemize}
\end{frame}
